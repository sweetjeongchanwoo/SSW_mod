#ドイツ内戦の終結
country_event = {
	id = ssw.24
	hidden = yes
	trigger = {
		has_global_flag = GER_civil_war
		OR = {
			AND = {
				country_exists = GER
				NOT = {
					country_exists = LTV
					country_exists = HLH
					country_exists = HGR
				}
			}
			AND = {
				country_exists = LTV
				NOT = {
					country_exists = GER
					country_exists = HLH
					country_exists = HGR
				}
			}
			AND = {
				country_exists = HLH
				NOT = {
					country_exists = GER
					country_exists = LTV
					country_exists = HGR
				}
			}
			AND = {
				country_exists = HGR
				NOT = {
					country_exists = GER
					country_exists = HLH
					country_exists = LTV
				}
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = { days = 1 }
	immediate = { set_global_flag = GER_civil_war_end }
	option = { name = ssw.24.a }
}
